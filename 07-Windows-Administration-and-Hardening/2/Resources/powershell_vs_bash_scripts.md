```bash
#!/bin/bash
check_number=2
if [ $check_number -eq 2 ]
then
    echo "Variable is equal to 2!"
else
    echo "Variable is NOT equal to 2!"
fi
```

```PowerShell
$checkNumber=2;
if ( $checkNumber -eq 2 ) {
    Write-Host "Variable is equal to 2!"
}
else {
    Write-Host "Variable is NOT equal to 2!"
}
```
